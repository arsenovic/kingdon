import numpy as np
from kingdon import Algebra
exp = lambda x: x.exp()
#from kingdon.numerical import exp as exp_num # work around
#exp = lambda x: exp_num(x, 30)
pga = Algebra(3,0,1,start_index=0) 
locals().update(pga.blades)


# parameters
thetas = np.linspace(0,1*np.pi,401)

r1,r2 = .2,.5
p ,q  = 2,3

Rs = [exp(-theta*e12*p)*exp(r2*e01)*exp(-theta*e13*q)*exp(r1*e01) for theta in thetas]

o      = e0.dual() # origin 
points = [k>>o for k in Rs]
pga.graph(
      *points ,
    grid=False, lineWidth=4,
)
